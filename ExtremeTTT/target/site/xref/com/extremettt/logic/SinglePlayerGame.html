11  /**
12   * Class contains the logic for a single player game.
13   *
14   * @author lalex
15   */
16  public class SinglePlayerGame extends TwoPlayerGame {
17  
18      private char whoseTurn = 'X';
19      private Frame frame;

21      private final char mode = 'S';

23      /**
24       * Constructor, same as parent's constructor.
25       */
26      public SinglePlayerGame() {
27          this.frame = new Frame(SinglePlayerGame.this);
28          initializeGameFrame();
29          this.cells = this.frame.getCells();
30  
31      }
32  
33      @Override
34      public char getMode() {
35          return mode;
36      }

38  }
