17  public class Start {
18      public Start(){
19      }
20      /**
21       * Starts a game depending on parameter
22       * <p
23       * prints "Single decided" if parameter is "S",
24       * prints "Two decided" if parameter is "T",
25       * @param String Should be "S" or "T"
26       */ 
27      public void startGame(String s) {
28          if (s.equals("S")) {
29              System.out.print("Single decided");
30              SinglePlayerGame sGame = new SinglePlayerGame();
31          } else if (s.equals("T")) {
32              System.out.print("Two decided");
33              TwoPlayerGame tGame = new TwoPlayerGame();
34          } else {
35              System.out.print("Problem in parameter");
36          }
37      }
38      /**
39       * asks user if single or two
40       * @return "S" or "T"
41       */
42      public String singleOrTwo() {
43          System.out.println("Single player or Two player game?(S / T)");
44          Scanner scanner = new Scanner(System.in);
45          String string = scanner.nextLine();
46          while (!string.equals("S") || !string.equals("T")) {
47              if (string.equals("S")) {
48                  return "S";
49              } else if (string.equals("T")) {
50                  return "T";
51              } else {
52                  System.out.println("not S nor T");
53                  string = scanner.nextLine();
54              }
55          }
56          return "Well something went wrong";
57      }
