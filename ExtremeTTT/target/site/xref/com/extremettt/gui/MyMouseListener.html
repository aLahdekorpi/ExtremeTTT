6   package com.extremettt.gui;
7   
8   import com.extremettt.logic.BoardSweeper;
9   import com.extremettt.logic.FullChecker;
10  import java.awt.event.MouseAdapter;
11  import java.awt.event.MouseEvent;
12  import com.extremettt.logic.TwoPlayerGame;
13  import com.extremettt.logic.WinChecker;
14  
15  /**
16   * Mouse Listener, what to do when clicking mouse.
17   *
18   * @author lalex
19   */
20  public class MyMouseListener extends MouseAdapter {
21  
22      private TwoPlayerGame game;
23      private char whoseTurn;
24      private char token;
25      private Cell cell;
26      private WinChecker winC;
27      private FullChecker fullC;
28      private BoardSweeper sweeper;
29  
30      /**
31       * Constructor.
32       *
33       * @param game Game
34       * @param cell Cell
35       */
36      public MyMouseListener(TwoPlayerGame game, Cell cell) {
37          this.game = game;
38          this.cell = cell;
39          this.token = cell.getToken();
40          this.whoseTurn = game.getWhoseTurn();
41          this.fullC = new FullChecker(game);
42          this.winC = new WinChecker(game);
43          this.sweeper = new BoardSweeper(game);
44      }
45  
46      /**
47       * What to do when clicking a cell.
48       *
49       * @param e Mouse Event
50       */
51      @Override
52      public void mouseClicked(MouseEvent e) {
53          if (this.cell.getToken() != ' ') {
54              return;
55          }
56          this.whoseTurn = this.game.getWhoseTurn();
57          //checks if game is full or won before setting any tokens
58          if (this.fullC.isFull() || this.winC.isWon()) {
59              this.game.getFrame().jlblStatus.setText("Game is already over");
60              return;
61          }
62          //if cell doesn't have a token and its someones turn, sets token to cell
63          if (this.cell.getToken() == ' ' && this.whoseTurn != ' ') {
64              this.cell.setToken(whoseTurn);
65              //nextTurn();
66          }
67  
68          //if game is won, declare winner
69          if (this.winC.isWon()) {
70              gameIsWonAfterClicked();
71              //if game is full, but game is not won, declare tie 
72          }
73          if (this.fullC.isFull() && !this.winC.isWon()) {
74              gameIsFullAfterClicked();
75          }
76          //if game is not won nor full, swap turns and print whose turn it is
77          if (!this.winC.isWon() && !this.fullC.isFull()) {
78              nextTurn();
79          }
80          this.game.getFrame().score.setText("X: " + this.game.getWinsX() + " O: " + this.game.getWinsO());
81      }
82  
83      /**
84       * Starts a new Game.
85       */
86      public void newGame() {
87          this.sweeper.nullifyGameBoard();
88      }
89  
90      /**
91       * Declares winner, sets whoseTurn to ' ', disposes frame, starts a new
92       * game.
93       */
94      public void gameIsWonAfterClicked() {
95          this.game.addPointToWinner();
96          newGame();
97          this.game.getFrame().jlblStatus.setText(this.whoseTurn + " won, game over");
98      }
99  
100     /**
101      * Declares tie, sets whoseTurn to ' ', disposes frame, starts a new game.
102      */
103     public void gameIsFullAfterClicked() {
104         this.game.getFrame().jlblStatus.setText("tie!");
105         newGame();
106     }
107 
108     /**
109      * Swaps turns, sets whoseTurn to correct char, sets jlbl text to whose turn
110      * is it.
111      */
112     public void nextTurn() {
113         this.game.swapTurns();
114         this.whoseTurn = this.game.getWhoseTurn();
115         this.game.getFrame().jlblStatus.setText(this.whoseTurn + "'s turn");
116     }
117 }
